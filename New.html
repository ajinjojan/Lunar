        // Moon phase calculation functions
        function getMoonPhase(date) {
            // Calculate moon phase (0 to 1)
            // 0 = New Moon, 0.25 = First Quarter, 0.5 = Full Moon, 0.75 = Last Quarter
           
            // Using a simplified calculation based on Julian date
            const year = date.getFullYear();
            const month = date.getMonth() + 1;
            const day = date.getDate();
           
            // Calculate Julian date
            let a = Math.floor((14 - month) / 12);
            let y = year + 4800 - a;
            let m = month + 12 * a - 3;
            let jd = day + Math.floor((153 * m + 2) / 5) + 365 * y + Math.floor(y / 4) - Math.floor(y / 100) + Math.floor(y / 400) - 32045;
           
            // Calculate days since known new moon (January 6, 2000)
            const knownNewMoon = 2451550.1;
            const daysSinceNewMoon = jd - knownNewMoon;
           
            // Calculate moon phase (0 to 1)
            const phase = (daysSinceNewMoon / 29.530588853) % 1;
           
            return phase;
        }

        function getPhaseName(phase) {
            if (phase < 0.03) return "New Moon";
            if (phase < 0.22) return "Waxing Crescent";
            if (phase < 0.28) return "First Quarter";
            if (phase < 0.47) return "Waxing Gibbous";
            if (phase < 0.53) return "Full Moon";
            if (phase < 0.72) return "Waning Gibbous";
            if (phase < 0.78) return "Last Quarter";
            return "Waning Crescent";
        }

        function getPhaseDetails(phase) {
            const names = {
                "New Moon": "The moon is not visible",
                "Waxing Crescent": "A thin crescent is visible",
                "First Quarter": "Half of the moon is illuminated",
                "Waxing Gibbous": "More than half is illuminated",
                "Full Moon": "The entire moon is visible",
                "Waning Gibbous": "The illumination is decreasing",
                "Last Quarter": "Half of the moon is illuminated",
                "Waning Crescent": "A thin crescent remains"
            };
            return names[getPhaseName(phase)];
        }

        function createMoonElement(phase, size = 'large') {
            const isLarge = size === 'large';
            const container = document.createElement('div');
            container.className = isLarge ? 'moon-container' : 'small-moon';
           
            const background = document.createElement('div');
            background.className = isLarge ? 'moon-background' : 'small-background';
           
            const phaseElement = document.createElement('div');
            phaseElement.className = isLarge ? 'moon-phase' : 'small-phase';
           
            // Calculate moon illumination
            let illumination;
            if (phase < 0.5) {
                // Waxing phase
                illumination = phase * 2;
            } else {
                // Waning phase
                illumination = (1 - phase) * 2;
            }
           
            // Set moon phase appearance
            if (phase < 0.03 || phase > 0.97) {
                // New moon - almost invisible
                phaseElement.style.background = 'transparent';
            } else if (phase < 0.5) {
                // Waxing phases (right side illuminated)
                phaseElement.style.clipPath = `inset(0 0 0 ${50 - illumination * 50}%)`;
                phaseElement.style.backgroundColor = '#f8fafc';
            } else {
                // Waning phases (left side illuminated)
                phaseElement.style.clipPath = `inset(0 ${50 - (1 - illumination) * 50}% 0 0)`;
                phaseElement.style.backgroundColor = '#f8fafc';
            }
           
            // Add glow for full moon
            if (phase > 0.47 && phase < 0.53) {
                phaseElement.classList.add('glow');
            }
           
            container.appendChild(background);
            container.appendChild(phaseElement);
           
            if (isLarge) {
                // Add craters for large moon
                const craters = [
                    { size: 15, top: 30, left: 50 },
                    { size: 20, top: 80, left: 120 },
                    { size: 12, top: 140, left: 70 },
                    { size: 18, top: 100, left: 40 },
                    { size: 10, top: 50, left: 130 }
                ];
               
                craters.forEach(crater => {
                    const craterEl = document.createElement('div');
                    craterEl.className = 'crater';
                    craterEl.style.width = crater.size + 'px';
                    craterEl.style.height = crater.size + 'px';
                    craterEl.style.top = crater.top + 'px';
                    craterEl.style.left = crater.left + 'px';
                    container.appendChild(craterEl);
                });
            }
           
            return container;
        }

        function updateTodayMoon() {
            const today = new Date();
            const phase = getMoonPhase(today);
            const phaseName = getPhaseName(phase);
            const phaseDetails = getPhaseDetails(phase);
           
            // Update date display
            document.getElementById('today-date').textContent = today.toLocaleDateString('en-US', {
                weekday: 'long',
                year: 'numeric',
                month: 'long',
                day: 'numeric'
            });
           
            // Update phase name and details
            document.getElementById('phase-name').textContent = phaseName;
            document.getElementById('phase-details').textContent = phaseDetails;
           
            // Update moon visualization
            const moonContainer = document.getElementById('today-moon');
            moonContainer.parentNode.replaceChild(createMoonElement(phase, 'large'), moonContainer);
        }

        function showNext7Days() {
            const container = document.getElementById('next-days');
            container.innerHTML = '';
            container.classList.remove('hidden');
           
            const today = new Date();
           
            for (let i = 1; i <= 7; i++) {
                const futureDate = new Date(today);
                futureDate.setDate(today.getDate() + i);
               
                const phase = getMoonPhase(futureDate);
                const phaseName = getPhaseName(phase);
               
                const dayElement = document.createElement('div');
                dayElement.className = 'moon-day';
               
                dayElement.innerHTML = `
                    <div class="text-center">
                        <div class="text-sm text-gray-400 mb-2">
                            ${futureDate.toLocaleDateString('en-US', { weekday: 'short' })}
                        </div>
                        <div class="text-xs text-gray-500 mb-3">
                            ${futureDate.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
                        </div>
                        ${createMoonElement(phase, 'small').outerHTML}
                        <div class="text-xs font-semibold mt-3 phase-label">${phaseName}</div>
                    </div>
                `;
               
                container.appendChild(dayElement);
            }
        }

        // Initialize the app
        document.addEventListener('DOMContentLoaded', function() {
            updateTodayMoon();

                    showNext7Days();

                } else {
                    container.classList.add('hidden');
                    button.textContent = 'Show Next 7 Days';
                }
            });

            const starsContainer = document.body;

                star.style.top = Math.random() * 100 + '%';
                star.style.animationDelay = Math.random() * 2 + 's';
                starsContainer.appendChild(star);
            }
        });
